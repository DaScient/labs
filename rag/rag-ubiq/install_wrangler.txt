# Make a project folder
mkdir rag-ubiq && cd rag-ubiq

# Initialize npm project for local wrangler binary
npm init -y

# Install wrangler as a dev dependency (no global install required)
# This will install Wrangler v2.x (preferred). Using npx to run it.
npm install --save-dev wrangler

# Optionally install prettier / eslint if you want linting later
npm install --save-dev prettier eslint
